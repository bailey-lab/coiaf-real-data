In this analysis file, we examine the result of running our novel algorithms on the Pf6 dataset. Additionally, we compare the results of our algorithm with the current state of the art model for predicting COI – THE REAL McCOIL.

We first load in our data.

# Path to data
path <- "~/Desktop/Malaria/COI data/"

# Read in the real data and the REAL McCOIL COI predictions
rmcl_wsafs <- readRDS(paste0(path, "RMCL_wsafs_unique.rds"))
rmcl_coi_out <- readRDS(paste0(path, "RMCL_coi_out.rds")) %>%
  tibble::as_tibble() %>%
  dplyr::mutate(dplyr::across(c(file, name), as.character)) %>%
  dplyr::rename(rmcl = COI) %>%
  dplyr::rename(rmcl_025 = COI_025) %>%
  dplyr::rename(rmcl_975 = COI_975)
complete_predictions <- readRDS(here::here(
  "data-outputs",
  "freq_variation.rds"
))

We want to make sure that we have tested all the patients, so we cross check our results with the results from THE REAL McCOIL and the initial data matrices.

# According to our predictions
length(unique(complete_predictions$name))
#> [1] 5970

# According to rmcl output
length(unique(rmcl_coi_out$name))
#> [1] 5157

# According to data matrices
purrr::map(seq_len(length(rmcl_wsafs)), ~ rownames(rmcl_wsafs[[.x]])) %>%
  unlist() %>%
  unique() %>%
  length()
#> [1] 5970

Looking at these values, we can see that for our data matrices and our predictions the number of patients match. However, THE REAL McCOIL output has fewer total patients. We need to be sure to exclude these when we are comparing our two methods. We, therefore, end up with r length(unique(rmcl_coi_out$name)) samples.

exclude <- setdiff(unique(complete_predictions$name), unique(rmcl_coi_out$name))
compare_data <- complete_predictions %>%
  dplyr::filter(!name %in% exclude) %>%
  dplyr::filter(rmcl_med != 25) %>%
  dplyr::relocate(Region, .after = dplyr::last_col())

Evaluate predictions

We first reshape our data so that we have a single column with our COI estimates. This involves creating two new additional column to represent the coi_method and optimization method we used to compute the COI.

pivot_data <- compare_data %>%
  tidyr::pivot_longer(
    cols = dplyr::starts_with(c("dis", "cont")),
    names_to = c(".value", "coi_method"),
    names_pattern = "(.*)_(.*)_.*"
  ) %>%
  tidyr::pivot_longer(
    cols = dplyr::any_of(c("dis", "cont")),
    names_to = "optimization",
    values_to = "COI",
    values_drop_na = TRUE
  ) %>%
  dplyr::relocate(
    dplyr::any_of(c("coi_method", "optimization", "COI")),
    .after = name
  )

We can next plot all our estimates and further include the deviation from the median THE REAL McCOIL estimation.

World maps

We need to access the Pf6 metadata in order to create a world map as the metadata contains information about the latitude and longitude of each sample.

meta <- readRDS(here::here("metadata", "pf6_meta.rds"))

patient_lat_long <- dplyr::left_join(
  compare_data, meta,
  by = c("name" = "Sample")
)

We can then generate several world maps with our data!

# Get average of data
map_average <- patient_lat_long %>%
  dplyr::group_by(Site, Lat, Long) %>%
  dplyr::summarise(
    coi_mean = mean(dis_var_med),
    coi_med = median(dis_freq_med),
    prev = median(as.numeric(prev_2_10)),
    rmcl_med = median(rmcl_med),
    .groups = "drop"
  ) %>%
  dplyr::rename(lat = Lat) %>%
  dplyr::rename(long = Long)

# Plot the mean COI
mean_world <- world_map(
  map_average,
  coi_mean,
  label = "Mean COI",
  alpha = 0.7,
  breaks = c(1.0, 1.5, 2.0, 2.5)
)
mean_world


# Plot the median COI
med_world <- world_map(
  map_average,
  coi_med,
  label = "Median COI",
  alpha = 0.7,
  breaks = c(1.0, 1.5, 2.0)

med_world

Prevalence & FwS

We can also look at the relationship between our COI estimations and the prevalence of mutations as well as the FwS metric.

prev_fws <- patient_lat_long %>%
  tidyr::pivot_longer(dplyr::any_of(c("cont_var_med", "rmcl_med")),
    names_to = "Method",
    values_to = "COI",
    values_drop_na = TRUE
  ) %>%
  dplyr::select(name, Region:Long, prev_2_10:COI) %>%
  dplyr::mutate(
    prev = as.numeric(prev_2_10),
    Method = dplyr::recode(
      Method,
      cont_var_med = "Continuous Method 1",
      rmcl_med = "THE REAL McCOIL"
    ),
    .keep = "unused"
  )

grouped <- prev_fws %>%

    COI = median(COI),

We could also look at the prevalence of mutations slightly differently and look at the ridge plot.

filter_data <- patient_lat_long %>%
  tidyr::pivot_longer(
    cols = dplyr::starts_with(c("dis", "cont")),
    names_to = c(".value", "coi_method"),
    names_pattern = "(.*)_(.*)_.*"
  ) %>%
  tidyr::pivot_longer(
    cols = dplyr::any_of(c("dis", "cont")),
    names_to = "optimization",
    values_to = "COI",
    values_drop_na = TRUE
  ) %>%
  dplyr::relocate(
    dplyr::any_of(c("coi_method", "optimization", "COI")),
    .after = name
  ) %>%
  dplyr::filter(coi_method == "var" & optimization == "dis")

continents <- filter_data %>%
  dplyr::group_by(Region) %>%
  dplyr::mutate(med_prev = median(prev_2_10, na.rm = TRUE)) %>%
  dplyr::filter(med_prev != 0) %>%
  dplyr::arrange(prev_2_10) %>%
  dplyr::ungroup() %>%
  dplyr::mutate(
    Region = forcats::as_factor(Region),
    continent = countrycode::countrycode(Country, "country.name", "continent")
  )

sorted_prev <- continents %>%
  dplyr::select(Region, med_prev) %>%
  dplyr::distinct() %>%
  dplyr::pull(med_prev) %>%
  sort()

ridges <- ggplot(
  data = continents,
  aes(y = forcats::fct_reorder(Region, med_prev), x = COI, fill = continent)
) +
  ggridges::geom_density_ridges() +
  ggridges::theme_ridges() +
  theme_coiaf() +
  scale_fill_viridis_d(name = "Continent") +
  annotate(
    "text",
    x = 6.25,
    y = 1:16 + 0.3,
    label = round(sorted_prev, 2),
    size = 2.5
  ) +
  labs(x = "COI", y = "Region")
